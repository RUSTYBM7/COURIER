#!/bin/bash

set -e

echo "🚀 Starting fresh project deployment..."

# ================================
# 1. PURGE ALL REPLIT ARTIFACTS
# ================================
echo "🧹 Removing Replit dependencies..."

rm -rf .replit
rm -rf replit.nix
rm -rf .config
rm -rf .cache
rm -rf .upm
rm -rf .pythonlibs

npm uninstall @replit/database @replit/object-storage @replit/agent replit 2>/dev/null || true

# ================================
# 2. CLEAN INSTALL STATE
# ================================
echo "🧽 Resetting dependency state..."

rm -rf node_modules
rm -rf package-lock.json
rm -rf yarn.lock
rm -rf pnpm-lock.yaml
rm -rf .vercel

npm cache clean --force

# ================================
# 3. INIT GIT
# ================================
echo "🔧 Initializing git..."

git init
git branch -M main

git add .
git commit -m "Initial production deploy" || true

# ================================
# 4. CONNECT YOUR NEW GITHUB REPO
# ================================
echo "🔗 Add your new GitHub repo"

echo ""
echo "Run this after script:"
echo "git remote add origin https://github.com/YOUR_USERNAME/YOUR_NEW_REPO.git"
echo "git push -u origin main"
echo ""

# ================================
# 5. INSTALL VERCEL
# ================================
echo "⬇️ Installing Vercel CLI..."

npm install -g vercel@latest

# ================================
# 6. AUTH
# ================================
export VERCEL_TOKEN="YOUR_VERCEL_TOKEN"

# ================================
# 7. LINK NEW PROJECT
# ================================
echo "🔗 Linking Vercel project..."

vercel link --yes --token=$VERCEL_TOKEN

# ================================
# 8. INSTALL
# ================================
echo "📦 Installing packages..."

if [ -f yarn.lock ]; then
  yarn install
elif [ -f pnpm-lock.yaml ]; then
  pnpm install
else
  npm install
fi

# ================================
# 9. BUILD
# ================================
echo "🏗️ Building..."

npm run build || echo "No build script found"

# ================================
# 10. FORCE DEPLOY
# ================================
echo "🚀 Deploying to production..."

vercel deploy --prod --force --yes --token=$VERCEL_TOKEN

echo "✅ New project deployed successfully"
